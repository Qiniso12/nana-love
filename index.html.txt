/* Buttons */
let noScale = 1;
document.getElementById('noBtn').onclick = () => {
    noScale -= 0.15;
    document.getElementById('noBtn').style.transform = `scale(${noScale})`;
    document.getElementById('yesBtn').style.transform = `scale(${1 + (1 - noScale)})`;
    if (noScale <= 0.2) document.getElementById('noBtn').remove();
};

document.getElementById('yesBtn').onclick = () => {
    document.getElementById('main').classList.add('hidden');
    document.getElementById('choice').classList.remove('hidden');
};

/* Navigation */
function openGallery() {
    hideAll();
    document.getElementById('gallery').classList.remove('hidden');
}

function openNote() {
    hideAll();
    document.getElementById('note').classList.remove('hidden');
}

function backToFrames() {
    hideAll();
    document.getElementById('choice').classList.remove('hidden');
}

function hideAll() {
    document.querySelectorAll('.container').forEach(c => c.classList.add('hidden'));
}
